## TDD 流程

### 写集成测试

```rust
#[test]
fn us18_create_with_category() {
    let db = setup_test_db();
    create_note(&db, "工作笔记", "内容", "工作");
    let notes = list_notes(&db);
    assert_eq!(notes[0].category, "工作");
}

#[test]
fn us19_add_tag() {
    let db = setup_test_db();
    create_note(&db, "笔记", "", "default");
    add_tag(&db, 1, "重要");
    let tags = get_tags(&db, 1);
    assert!(tags.contains(&"重要".to_string()));
}

#[test]
fn us20_filter_by_category() {
    let db = setup_test_db();
    create_note(&db, "A", "", "工作");
    create_note(&db, "B", "", "个人");
    let filtered = list_notes_by_category(&db, "工作");
    assert_eq!(filtered.len(), 1);
}

#[test]
fn us21_filter_by_tag() {
    let db = setup_test_db();
    create_note(&db, "A", "", "default");
    add_tag(&db, 1, "重要");
    let filtered = list_notes_by_tag(&db, "重要");
    assert_eq!(filtered.len(), 1);
}

#[test]
fn us22_pagination() {
    let db = setup_test_db();
    for i in 1..=25 { create_note(&db, &format!("笔记{}", i), "", "default"); }
    let page1 = list_notes_page(&db, 1, 20);
    let page2 = list_notes_page(&db, 2, 20);
    assert_eq!(page1.len(), 20);
    assert_eq!(page2.len(), 5);
}
```
